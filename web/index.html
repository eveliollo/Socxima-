/* ── MINI SPARKLINE CHARTS (VERSIÓN PREMIUM GLOW) ── */ function drawSparkline(canvasId, data, color) { const canvas = document.getElementById(canvasId); if (!canvas || data.length < 2) return; const ctx = canvas.getContext('2d'); // Multiplicamos por el ratio de píxeles del dispositivo para un renderizado nítido (Retina/4K) const dpr = window.devicePixelRatio || 1; const rect = canvas.getBoundingClientRect(); canvas.width = rect.width * dpr; canvas.height = rect.height * dpr; ctx.scale(dpr, dpr); const W = rect.width; const H = rect.height; const min = Math.min(...data); const max = Math.max(...data); const range = max - min || 1; ctx.clearRect(0, 0, W, H); // 1. DIBUJAR RETÍCULA TECNOLÓGICA DE FONDO (Grid lines) ctx.strokeStyle = 'rgba(10, 32, 53, 0.3)'; ctx.lineWidth = 0.5; // Líneas horizontales for (let i = 1; i < 4; i++) { ctx.beginPath(); ctx.moveTo(0, (H / 4) * i); ctx.lineTo(W, (H / 4) * i); ctx.stroke(); } // Líneas verticales for (let i = 1; i < 6; i++) { ctx.beginPath(); ctx.moveTo((W / 6) * i, 0); ctx.lineTo((W / 6) * i, H); ctx.stroke(); } // Mapeo de puntos const points = data.map((v, i) => { return { x: (i / (data.length - 1)) * W, y: H - ((v - min) / range) * H * 0.75 - H * 0.12 // Margen superior/inferior profesional }; }); // 2. GRADIENTE DE ÁREA DEBAJO DE LA LÍNEA const areaGrad = ctx.createLinearGradient(0, 0, 0, H); areaGrad.addColorStop(0, color + '33'); // Opacidad del 20% arriba areaGrad.addColorStop(0.5, color + '0D'); // Opacidad del 5% al medio areaGrad.addColorStop(1, 'transparent'); // Desvanecido total abajo ctx.beginPath(); ctx.moveTo(0, H); points.forEach((p, i) => { if (i === 0) ctx.lineTo(p.x, p.y); else { // Curvatura suave (Bezier) para evitar esquinas rígidas y dar fluidez líquida const cpX = (points[i - 1].x + p.x) / 2; ctx.bezierCurveTo(cpX, points[i - 1].y, cpX, p.y, p.x, p.y); } }); ctx.lineTo(W, H); ctx.closePath(); ctx.fillStyle = areaGrad; ctx.fill(); // 3. EFECTO DE RESPLANDOR (GLOW) BAJO LA LÍNEA PRINCIPAL ctx.shadowColor = color; ctx.shadowBlur = 10; ctx.shadowOffsetX = 0; ctx.shadowOffsetY = 0; // 4. LÍNEA DE TENDENCIA PRINCIPAL ctx.beginPath(); points.forEach((p, i) => { if (i === 0) ctx.moveTo(p.x, p.y); else { const cpX = (points[i - 1].x + p.x) / 2; ctx.bezierCurveTo(cpX, points[i - 1].y, cpX, p.y, p.x, p.y); } }); ctx.strokeStyle = color; ctx.lineWidth = 2.5; ctx.lineCap = 'round'; ctx.lineJoin = 'round'; ctx.stroke(); // Desactivar sombra para que no afecte a otros elementos del renderizado ctx.shadowBlur = 0; // 5. PUNTO DE PRECIO ACTUAL (ÚLTIMO DATO) CON PULSACIÓN DE NEÓN const lastP = points[points.length - 1]; // Aura exterior semitransparente (Efecto radar) ctx.beginPath(); ctx.arc(lastP.x - 3, lastP.y, 6, 0, Math.PI * 2); ctx.fillStyle = color + '44'; ctx.fill(); // Núcleo sólido brillante ctx.beginPath(); ctx.arc(lastP.x - 3, lastP.y, 3, 0, Math.PI * 2); ctx.fillStyle = '#ffffff'; // Centro blanco para simular máxima intensidad luminosa ctx.strokeStyle = color; ctx.lineWidth = 1.5; ctx.fill(); ctx.stroke(); } ctx.stroke(); } }
